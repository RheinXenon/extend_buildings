# Building cap for Urban-Rural Junction
# Based on local population

urban_rural_junction_cap = {
	add = {
		desc = "BUILDING_LEVEL_BASE"
		value = 0  # No base level, purely population dependent
	}
	add = {
		desc = "BUILDING_LEVEL_POPULATION"
		value = location_population
		divide = 5000  # 1 level per 5000 population
	}
	# Ensure at least 1 level if population requirement is met
	if = {
		limit = {
			location_population >= 1000
		}
		add = {
			desc = "BUILDING_LEVEL_MINIMUM"
			value = 1
		}
	}
}
