# Urban-Rural Junction Building
# 城乡结合部建筑
# This building increases the building slots limit in cities and towns

urban_rural_junction = {
	is_foreign = no
	
	max_levels = {
		add = {
			desc = "BUILDING_LEVEL_BASE"
			value = 1
		}
		add = {
			desc = "BUILDING_LEVEL_DEVELOPMENT"
			value = development
			multiply = 0.1
		}
		add = {
			desc = "BUILDING_LEVEL_POPULATION"
			value = population
			multiply = 0.05
		}
	}
	
	pop_type = peasants

	category = infrastructure_category
	employment_size = tiny_rural_peasant_produce_employment  # tiny_rural_peasant_produce_employment = 250 workers per level
	
	# Only available in cities and towns, not rural settlements
	rural_settlement = no
	town = yes
	city = yes
	
	build_time = infrastructure_build_time

	# Single production method for maintenance
	possible_production_methods = {
		urban_rural_junction_maintenance
	}

	construction_demand = town_building_construction

	# Each level adds 10 free building levels (building slots)
	modifier = {
		free_building_levels = 10
	}
}
